\section{Supplementary Proofs}
\label{app:proofs}

\subsection{Detailed Expansion: Coupling Terms in Lemma~\ref{lem:stability}}
\label{app:step5}

We provide the expansion of term (ii) from \eqref{eq:coupling_split} and show
how the non-vacuum condition (\Cref{ass:bound}) and the
\(L_{p\rho}\)-control of \(\nabla_pR\) lead to the conservative rate
\(\kappa_\eps\) used in \Cref{thm:main}.

With $R^i(x,p) \coloneqq R(x,\bar\rho_i(x,t),p)$, the $p$-variation contribution is:
\begin{align*}
  \text{(ii)}
  &= \eps\int_\Omega \bigl[R^1(x,p_1) - R^1(x,p_2)\bigr]\delta\rho\,dx
   - \eps\int_\Omega \delta p\cdot\bigl[\rho_1\nabla_pR^1(x,p_1)
      - \rho_2\nabla_pR^2(x,p_2)\bigr]\,dx.
\end{align*}

\noindent\textbf{Bounding term (ii).}
By the mean value theorem and \Cref{ass:lip}, $|R^1(p_1) - R^1(p_2)| \leq L_{R,p}|\delta p|$.
For the current part, write $\rho_1\nabla_pR^1(p_1) - \rho_2\nabla_pR^2(p_2) =
\rho_1[\nabla_pR^1(p_1)-\nabla_pR^1(p_2)] + \rho_1[\nabla_pR^1(p_2)-\nabla_pR^2(p_2)]
+ \delta\rho\cdot\nabla_pR^2(p_2)$.
Using $|\nabla_p^2 R| \leq L_{R,p}$,
$|\nabla_pR^1(p_2)-\nabla_pR^2(p_2)| \leq L_{p\rho}|\delta\bar\rho|$,
and \Cref{ass:bound} ($\rho_i \leq \rho_{\max}$):
\begin{equation}\label{eq:p_var_bound}
  |\text{(ii)}|
  \leq 3\,\eps\,L_{R,p}\,\rho_{\max}\int_\Omega |\delta p|^2\,dx
  + \eps\,L_{p\rho}\rho_{\max}
    \norm{\delta p}_{L^2}\norm{\delta\bar\rho}_{L^2}.
\end{equation}

\noindent\textbf{Why the absorption is rigorous.}
The strong convexity bound \eqref{eq:convex_bound} gives
$A_0+C_0 \leq -(c_0/2)\int(\rho_1+\rho_2)|\delta p|^2\,dx$.
The non-vacuum condition $\rho_1+\rho_2 \geq 2\rho_{\min}$ then yields
\[
  A_0+C_0 \;\leq\; -c_0\,\rho_{\min}\int_\Omega |\delta p|^2\,dx.
\]
Set
\[
  \alpha_\eps=c_0\rho_{\min}-3\eps L_{R,p}\rho_{\max}>0.
\]
Then the momentum part is bounded by
\[
  A_0+C_0+\eps(A_R+C_R)_{\text{(ii)}}
  \leq -\alpha_\eps\norm{\delta p}_{L^2}^2
  + \eps L_{p\rho}\rho_{\max}
    \norm{\delta p}_{L^2}\norm{\delta\bar\rho}_{L^2}.
\]
Maximizing the right-hand side over $\norm{\delta p}_{L^2}$ gives
\[
  B_\eps\norm{\delta\bar\rho}_{L^2}^2,\qquad
  B_\eps=\frac{\eps^2L_{p\rho}^2\rho_{\max}^2}{4\alpha_\eps}.
\]
Combining this term with term (i) of \eqref{eq:rho_bound} gives the stability
bound \eqref{eq:stability}.

\subsection{Explicit Verification for the Bilinear Case}
\label{app:bilinear}

For $R(x,\rho,p) = \rho\cdot p$ (MFG-LWR traffic Hamiltonian), we verify
the coupling bound by direct computation.

Here $H^i(x,p) = H_0(x,p) + \eps\,\bar\rho_i\cdot p$, so
$\nabla_p H^i(p) = \nabla_pH_0(p) + \eps\bar\rho_i$.

\paragraph{Base part.} With $H_0(p) = \frac{1}{2}|p|^2 - p$ (so $c_0=1$):
\begin{align*}
  A_0 + C_0
  &= \int_\Omega\rho_1\bigl[H_0(p_1) - H_0(p_2)
      - \nabla_pH_0(p_1)\cdot\delta p\bigr]dx
    -\int_\Omega\rho_2\bigl[H_0(p_1) - H_0(p_2)
      - \nabla_pH_0(p_2)\cdot\delta p\bigr]dx\\
  &= -\int_\Omega\rho_1\tfrac{1}{2}|\delta p|^2\,dx
     -\int_\Omega\rho_2\tfrac{1}{2}|\delta p|^2\,dx
   = -\frac{1}{2}\int_\Omega(\rho_1+\rho_2)|\delta p|^2\,dx \;\leq\; 0. \checkmark
\end{align*}

\paragraph{Coupling part.} Expanding $\eps(A_R+C_R)$ with $R=\bar\rho\cdot p$:
\begin{align*}
  \eps(A_R+C_R)
  &= \eps\int_\Omega(\bar\rho_1 p_1 - \bar\rho_2 p_2)\delta\rho\,dx
   - \eps\int_\Omega\delta p\cdot(\rho_1\bar\rho_1 - \rho_2\bar\rho_2)\,dx.
\end{align*}
Expanding and collecting:
\[
  = \eps\int_\Omega(\bar\rho_1-\bar\rho_2)p_1\rho_1\,dx
   +\eps\int_\Omega\bar\rho_2(p_1-p_2)\rho_1\,dx
   -\eps\int_\Omega(\bar\rho_1-\bar\rho_2)p_1\rho_2\,dx
   +\cdots
\]
After cancellation, only the cross-term survives:
\[
  \eps(A_R+C_R) = -\eps\int_\Omega\delta\bar\rho\,
  \bigl[(p_1-p_2)\rho_2 - p_2\,\delta\rho\bigr]\,dx.
\]
Bounding each piece with \Cref{ass:bound}:
\begin{align*}
  |\eps(A_R+C_R)|
  &\leq \eps\,u_{\max}\norm{\delta\bar\rho}_{L^2}\norm{\delta\rho}_{L^2}
   + \eps\int_\Omega |\delta\bar\rho|\,\rho_2\,|\delta p|\,dx.
\end{align*}

\noindent\textbf{Absorption of the $\delta p$-term.}
By Cauchy-Schwarz with weight $\rho_2$:
\[
  \int_\Omega |\delta\bar\rho|\,\rho_2\,|\delta p|\,dx
  \;\leq\;
  \Bigl(\int_\Omega \rho_2\,|\delta\bar\rho|^2\,dx\Bigr)^{1/2}
  \Bigl(\int_\Omega \rho_2\,|\delta p|^2\,dx\Bigr)^{1/2}
  \;\leq\;
  \rho_{\max}^{1/2}\,\norm{\delta\bar\rho}_{L^2}\cdot
  \Bigl(\tfrac{1}{2}J\Bigr)^{1/2},
\]
where $J = \int_\Omega(\rho_1+\rho_2)|\delta p|^2\,dx$.
By Young's inequality ($ab \leq \frac{c_0}{4}a^2 + \frac{1}{c_0}b^2$):
\[
  \eps\,\rho_{\max}^{1/2}\norm{\delta\bar\rho}_{L^2}\cdot\tfrac{J^{1/2}}{\sqrt 2}
  \;\leq\; \frac{c_0}{4}\,J + \frac{\eps^2\rho_{\max}}{c_0}\,\norm{\delta\bar\rho}_{L^2}^2.
\]
The non-vacuum condition gives $A_0+C_0 \leq -(c_0/2)J$, so:
\[
  A_0+C_0 + \eps(A_R+C_R)
  \leq -\frac{c_0}{4}\,J
  + \eps\,u_{\max}\norm{\delta\bar\rho}_{L^2}\norm{\delta\rho}_{L^2}
  + \frac{\eps^2\rho_{\max}}{c_0}\,\norm{\delta\bar\rho}_{L^2}^2.
\]
Inserting into the balance identity \eqref{eq:balance} and dropping
$-(c_0/4)J \leq 0$, then dividing by $\norm{\delta\rho}_{L^2}$:
\[
  \lambda\,\norm{\delta\rho}_{L^2}
  \;\leq\;
  \eps\,u_{\max}\,\norm{\delta\bar\rho}_{L^2}
  \;+\; \frac{\eps^2\rho_{\max}}{c_0}\,
        \frac{\norm{\delta\bar\rho}_{L^2}^2}{\norm{\delta\rho}_{L^2}}.
\]
The leading term $\eps u_{\max}/\lambda$ is the first-order contribution to
the local rate.  The second term is exactly the frozen-velocity correction that
is represented by \(B_\eps\) in the general theorem.  In the density-only case
this correction vanishes, recovering the clean rate
\(\eps L_R/\lambda\). $\hfill\square$

\subsection{Proof of Proposition~\ref{prop:inexact}}

By triangle inequality and Theorem~\ref{thm:main}:
\begin{align*}
  \norm{\rho^{k+1}_N - \rho^*}_{L^2}
  &\leq \norm{\rho^{k+1}_N - \rho^{k+1}}_{L^2}
   + \norm{\rho^{k+1} - \rho^*}_{L^2}
  \leq \delta_N + \kappa_\eps\,\norm{\rho^k_N - \rho^*}_{L^2}.
\end{align*}
Iterating and summing
$\sum_{j=0}^{k-1}\kappa_\eps^j \leq \frac{1}{1-\kappa_\eps}$
yields the result. $\hfill\square$

\subsection{On the Equivalence of $L^2$ and $\mathcal{W}_2$ Convergence}
\label{app:wasserstein}

For probability densities $\rho_1,\rho_2$ on bounded $\Omega$ with
$\norm{\rho_i}_{L^\infty} \leq \rho_{\max}$:
\[
  \mathcal{W}_2(\rho_1,\rho_2)^2
  \;\leq\; \mathrm{diam}(\Omega)\cdot\norm{\rho_1-\rho_2}_{L^1}
  \;\leq\; \mathrm{diam}(\Omega)\cdot|\Omega|^{1/2}\cdot\norm{\rho_1-\rho_2}_{L^2}.
\]
See \citet{villani2009optimal}, Theorem~7.12.
Hence $L^2$ convergence at rate $\kappa_\eps^k$ implies $\mathcal{W}_2$
convergence at rate $\kappa_\eps^{k/2}$ through this elementary estimate
(up to domain-dependent constants). $\hfill\square$
