\title{A GAN-Compatible Fixed-Point Method for Mean Field Games\\
with Weakly Non-Separable Hamiltonians}

\author{Anonymous Authors}

\begin{document}

\maketitle

\begin{abstract}
GAN-based solvers for mean field games (MFGs) learn population densities
distributionally, but existing formulations rely on separable Hamiltonians.
Many models of interest, including traffic-flow MFGs, contain non-separable
density--momentum couplings.  We introduce a weakly non-separable class
\(H(x,\rho,p)=H_0(x,p)-f_0(x,\rho)+\eps R(x,\rho,p)\) and study a fixed-point
outer iteration that freezes the density inside the non-separable residual.
Each outer step is then a separable MFG subproblem and can be solved by an
existing GAN-based inner method.  Under Lasry--Lions monotonicity, strong
convexity of the base Hamiltonian, bounded densities, and bounded momentum, we
prove that the exact outer map is a contraction in \(L^2\) with explicit rate
\(\kappa=\eps L_R/\lambda\).  This implies convergence in \(\W_2\) with an
explicit induced rate.  We also show that the threshold is sharp for a
linearized ergodic family on the torus, and provide numerical experiments that
validate the predicted linearized phase transition.
\end{abstract}

\section{Introduction}

Mean field games (MFGs) model equilibria in large populations of rational
agents through a backward Hamilton--Jacobi--Bellman equation coupled to a
forward Fokker--Planck equation:
\begin{align}
  -\partial_t \varphi-\nu\Delta\varphi+H(x,\rho,\nabla\varphi)&=0,
  \label{eq:hjb}\\
  \partial_t\rho-\nu\Delta\rho-\Div\!\bigl(\rho\nabla_pH(x,\rho,\nabla\varphi)\bigr)&=0.
  \label{eq:fp}
\end{align}
When \(H(x,\rho,p)=H_0(x,p)-f_0(x,\rho)\), the system admits a variational
structure connected to Wasserstein GANs, enabling adversarial solvers such as
APAC-Net and MFGAN.  The separability assumption, however, excludes important
models where density and momentum are coupled directly.

This paper asks whether the GAN-compatible structure can be retained for a
controlled class of non-separable Hamiltonians.  Our observation is simple:
if the density appearing in the non-separable part is temporarily frozen, the
remaining subproblem is separable in the sense required by existing GAN-based
MFG solvers.  We therefore separate the analysis into an outer fixed-point map,
which handles the non-separable coupling, and an inner separable solver, which
may be implemented by any GAN-based MFG method.

\paragraph{Contributions.}
We make four contributions.  First, we define weakly non-separable
Hamiltonians and the corresponding GAN-tractable regime.  Second, we propose
Extended-MFGAN, a fixed-point wrapper around separable GAN-based MFG solvers.
Third, we prove an explicit \(L^2\) contraction theorem for the exact outer map
and derive Wasserstein convergence as a corollary.  Fourth, we show sharpness
of the threshold in a linearized ergodic family and validate the predicted
phase transition numerically.

\section{Weakly Non-Separable Hamiltonians}

\begin{definition}
A Hamiltonian is weakly non-separable with coupling strength \(\eps\geq0\) if
\begin{equation}
  H(x,\rho,p)=H_0(x,p)-f_0(x,\rho)+\eps R(x,\rho,p),
  \label{eq:decomp_neurips}
\end{equation}
where \(H_0\) is convex in \(p\), \(f_0\) is monotone in \(\rho\), and
\(R\) is the residual density--momentum coupling.
\end{definition}

The example motivating the paper is the one-parameter traffic-flow family
\begin{equation}
  H_\eps(\rho,p)=\frac12 p^2-p+\eps \rho p,
  \label{eq:traffic_family}
\end{equation}
whose case \(\eps=1\) corresponds to the common normalization
\(\frac12 p^2-(1-\rho)p\).  For fixed \(\bar\rho\), the effective Hamiltonian
\(H_0(p)+\eps R(\bar\rho,p)\) no longer depends on the unknown density and is
therefore a separable MFG subproblem.

\section{Extended-MFGAN}

Let \(\bar\rho\) denote the density frozen from the previous outer step.  Define
\[
  H^{\bar\rho}(x,p)=H_0(x,p)+\eps R(x,\bar\rho(x,t),p).
\]
The outer fixed-point map \(\mathcal{T}\) sends \(\bar\rho\) to the density
\(\rho\) obtained by solving the separable MFG with Hamiltonian
\(H^{\bar\rho}\) and interaction \(f_0\).

\begin{algorithm}[t]
\caption{Extended-MFGAN}
\label{alg:neurips}
\begin{algorithmic}[1]
\Require decomposition \(H=H_0-f_0+\eps R\), initial density \(\rho_0\),
terminal cost \(g\), outer steps \(K\), inner steps \(N\)
\State initialize \(\rho^0\gets \rho_0\)
\For{\(k=0,\ldots,K-1\)}
  \State form \(H^k(x,p)=H_0(x,p)+\eps R(x,\rho^k(x,t),p)\)
  \State solve the separable MFG with Hamiltonian \(H^k\) using a GAN-based
  inner solver, e.g. APAC-Net:
  \[
    (\varphi^{k+1},\rho^{k+1})\gets \textsc{GAN-MFG}(H^k,f_0,g,\rho_0,N)
  \]
\EndFor
\State \Return \((\varphi^K,\rho^K)\)
\end{algorithmic}
\end{algorithm}

\section{Convergence of the Exact Outer Map}

We state the theorem for exact inner solves.  Approximation error from finite
inner training appears additively in the usual perturbed-contraction bound.

\begin{assumption}
\label{ass:neurips}
The following hold for all frozen densities considered by the iteration.
\begin{enumerate}
\item \(H_0\) is \(c_0\)-strongly convex in \(p\).
\item \(f_0\) is \(\lambda\)-strongly monotone:
\[
  \int (f_0(x,\rho_1)-f_0(x,\rho_2))(\rho_1-\rho_2)\,dx
  \geq \lambda\norm{\rho_1-\rho_2}_{L^2}^2 .
\]
\item Densities satisfy \(0<\rho_{\min}\leq \rho\leq\rho_{\max}\).
\item Momentum is bounded: \(\norm{\nabla\varphi_{\bar\rho}}_{L^\infty}\leq u_{\max}\).
\item On \(|p|\leq u_{\max}\), \(R\) is \(L_R\)-Lipschitz in \(\rho\), and
the \(p\)-variation terms are dominated by strong convexity:
\[
  C_R\eps L_{R,p}\rho_{\max}<c_0\rho_{\min},
\]
for the constant \(C_R\) appearing in the stability estimate.
\end{enumerate}
\end{assumption}

\begin{theorem}[Outer contraction]
\label{thm:l2_contraction_neurips}
Under \Cref{ass:neurips}, if \(\kappa=\eps L_R/\lambda<1\), then the exact
outer map \(\mathcal{T}\) is a contraction:
\[
  \norm{\mathcal{T}(\bar\rho_1)-\mathcal{T}(\bar\rho_2)}_{L^2((0,T)\times\Omega)}
  \leq
  \kappa\norm{\bar\rho_1-\bar\rho_2}_{L^2((0,T)\times\Omega)} .
\]
Consequently, the exact Extended-MFGAN outer iteration satisfies
\[
  \norm{\rho^k-\rho^\star}_{L^2}
  \leq
  \kappa^k\norm{\rho^0-\rho^\star}_{L^2}.
\]
\end{theorem}

\begin{proof}[Proof sketch]
Compare two separable subproblems with frozen densities \(\bar\rho_1\) and
\(\bar\rho_2\).  Let \(\delta\rho=\rho_1-\rho_2\),
\(\delta\bar\rho=\bar\rho_1-\bar\rho_2\), and
\(\delta p=\nabla\varphi_1-\nabla\varphi_2\).  The Lasry--Lions duality identity
for \(\frac{d}{dt}\int\delta\varphi\,\delta\rho\) gives
\[
  \lambda\norm{\delta\rho}_{L^2}^2
  \leq
  \eps L_R\norm{\delta\bar\rho}_{L^2}\norm{\delta\rho}_{L^2}
  + \mathcal{E}_p .
\]
The term \(\mathcal{E}_p\) contains the \(p\)-variation of the residual.  Strong
convexity of \(H_0\), the non-vacuum bound, and the final condition in
\Cref{ass:neurips} imply \(\mathcal{E}_p\leq0\).  Dividing by
\(\norm{\delta\rho}_{L^2}\) yields the contraction.  Banach's fixed-point
theorem gives the stated convergence to the unique fixed point.
\end{proof}

\begin{corollary}[Wasserstein convergence]
If \(\Omega\) is bounded and the densities are uniformly bounded, then
\[
  \W_2(\rho^k,\rho^\star)
  \leq
  C_\Omega \kappa^{k/2}\norm{\rho^0-\rho^\star}_{L^2}^{1/2}.
\]
\end{corollary}

\section{Sharpness in Linearized Ergodic MFGs}

The condition \(\eps L_R<\lambda\) is sharp for the linearized outer map.  On
\(\Omega=\T^d\), consider the ergodic density-only family
\[
  H(p,\rho,\bar\rho)=\frac12 |p|^2-\lambda\rho+\eps L_R\bar\rho .
\]
The uniform solution is \(\rho^\star\equiv1\), \(\nabla\varphi^\star=0\).
Linearizing the frozen-density map around this solution gives
\[
  -\lambda\delta\rho+\eps L_R\delta\bar\rho=0,
  \qquad
  \delta\rho=\frac{\eps L_R}{\lambda}\delta\bar\rho .
\]
Thus the local operator norm is exactly \(\eps L_R/\lambda\), so the threshold
cannot be improved for this family.

For the traffic family in \Cref{eq:traffic_family}, the no-drift equilibrium
satisfies \(p^\star=1-\eps\).  The linearized residual has effective Lipschitz
constant \(L_R(p^\star)=|1-\eps|\), giving the predicted local rate
\[
  \kappa_{\mathrm{traffic}}=\frac{\eps |1-\eps|}{\lambda}.
\]
For \(\lambda=0.2\), the equation \(\kappa_{\mathrm{traffic}}=1\) has two roots,
approximately \(0.276\) and \(0.724\), giving a two-sided phase transition.

\section{Discussion and Limitations}

The result should be interpreted as a contraction theorem for the exact outer
map, not as a complete optimization guarantee for GAN training.  Its value is
that it identifies a regime in which non-separable MFGs can be reduced to a
sequence of separable, GAN-compatible subproblems with an explicit rate.  A full
NeurIPS submission should add either a neural APAC-Net-style inner
implementation or clearly position the current experiments as theory-validation
experiments.  Important limitations are the non-vacuum condition, bounded
momentum, and the restriction to weak residual coupling.

\section{Perturbed Inner Solves}

If the inner solver returns \(\rho_N^{k+1}\) with
\(\norm{\rho_N^{k+1}-\mathcal{T}(\rho_N^k)}_{L^2}\leq\delta_N\), then
\[
  \norm{\rho_N^k-\rho^\star}_{L^2}
  \leq
  \kappa^k\norm{\rho^0-\rho^\star}_{L^2}
  +\frac{\delta_N}{1-\kappa}.
\]
This follows by iterating
\(\norm{\rho_N^{k+1}-\rho^\star}_{L^2}\leq
\delta_N+\kappa\norm{\rho_N^k-\rho^\star}_{L^2}\).
