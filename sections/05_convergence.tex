\section{Convergence Analysis}
\label{sec:convergence}

\subsection{Assumptions}

\begin{assumption}[Strongly convex base Hamiltonian]\label{ass:base}
$H_0 : \Omega \times \R^d \to \R$ is smooth and \emph{strongly} convex in $p$:
\[
  \nabla^2_{pp} H_0(x,p) \geq c_0\, I_d, \quad \forall\, (x,p) \in \Omega \times \R^d,
  \quad c_0 > 0.
\]
\end{assumption}

\begin{assumption}[Lasry-Lions monotonicity]\label{ass:mono}
$f_0 : \Omega \times \R \to \R$ satisfies
\[
  \int_\Omega \bigl(f_0(x,\rho_1) - f_0(x,\rho_2)\bigr)(\rho_1 - \rho_2)\,dx
  \;\geq\; \lambda\, \norm{\rho_1 - \rho_2}^2_{L^2(\Omega)},
  \qquad \lambda > 0,
\]
for all $\rho_1, \rho_2 \in L^2(\Omega)$.
\end{assumption}

\begin{assumption}[Bounded momentum]\label{ass:pmax}
There exists $u_{\max} < \infty$ such that the value functions
$\varphi_1, \varphi_2$ in \Cref{lem:stability} satisfy
$\norm{\nabla\varphi_i}_{L^\infty((0,T)\times\Omega)} \leq u_{\max}$.
\end{assumption}

\begin{assumption}[Lipschitz coupling on the admissible region]\label{ass:lip}
Let
$
\mathcal{D} \coloneqq
\Omega \times [\rho_{\min},\rho_{\max}] \times \{p \in \R^d : |p| \leq u_{\max}\}
$.
The coupling $R$ is Lipschitz on $\mathcal{D}$:
\[
  \sup_{(x,p):\, |p| \leq u_{\max}}\,
   |R(x,\rho_1,p) - R(x,\rho_2,p)| \leq L_R\, |\rho_1 - \rho_2|,
  \qquad
  \sup_{(x,\rho,p) \in \mathcal{D}}\, |\nabla_p R(x,\rho,p)| \leq L_{R,p}.
\]
\end{assumption}
\begin{remark}
\Cref{ass:pmax} is needed because $R$ may grow with $|p|$.
For the bilinear case $R(x,\rho,p) = \rho\cdot p$ one has
$L_R = u_{\max}$, since $|R(\rho_1,p)-R(\rho_2,p)| = |p||\rho_1-\rho_2|
\leq u_{\max}|\rho_1-\rho_2|$ on the admissible region.
For density-only $R(x,\rho)$ the bound on $|p|$ plays no role.
The boundedness of $\nabla\varphi$ is standard for viscosity solutions
of HJB equations with smooth coercive data
\citep{lasry2007mean,carmona2018probabilistic}.
\end{remark}

\begin{assumption}[Bounded densities]\label{ass:bound}
Solutions satisfy the \emph{non-vacuum condition}: there exist constants
$0 < \rho_{\min} \leq \rho_{\max} < \infty$ such that
\[
  \rho_{\min} \;\leq\; \rho(t,x) \;\leq\; \rho_{\max}
  \qquad \text{a.e.\ on } (0,T)\times\Omega.
\]
\end{assumption}
\begin{remark}
The lower bound $\rho_{\min} > 0$ rules out vacuum regions.
It is standard in congestion-type MFGs and in kinetic traffic models,
where a strictly positive background density is physically natural.
When $L_{R,p} = 0$ (i.e.\ $R$ is independent of $p$), only the
upper bound $\rho_{\max}$ is used and $\rho_{\min}$ plays no role.
\end{remark}

\begin{assumption}[Unique sub-problem solution]\label{ass:unique}
For each frozen $\bar\rho \in L^\infty$, the separable MFG with effective
Hamiltonian $H^{\bar\rho}(x,p) = H_0(x,p) + \eps R(x,\bar\rho,p)$
and interaction $f_0$ has a unique Nash equilibrium
$(\varphi_{\bar\rho},\rho_{\bar\rho}) \in W^{1,2}\times W^{1,2}$.
\end{assumption}

\subsection{The GAN-Tractable Regime}

\begin{definition}[GAN-tractable condition]\label{def:tractable_full}
A weakly non-separable Hamiltonian with parameters
$(c_0, \lambda, L_R, L_{R,p}, \rho_{\min}, \rho_{\max})$ is
\emph{GAN-tractable} if
\begin{equation}\label{eq:gan_tractable}
  \kappa \;\coloneqq\; \frac{\eps\, L_R}{\lambda}
  \;<\; 1
  \qquad \text{and} \qquad
  3\,\eps\, L_{R,p}\, \rho_{\max} \;<\; c_0\,\rho_{\min}.
\end{equation}
\end{definition}

\noindent The first condition controls the $\rho$-coupling strength relative to
monotonicity; it determines the convergence rate $\kappa$.
The second condition ensures that the $p$-variation of $R$ is dominated by
the strong convexity of $H_0$; it depends on the density contrast
$\rho_{\max}/\rho_{\min}$ and is vacuous when $L_{R,p} = 0$.
\begin{remark}
The factor of $3$ in the second condition arises from the three-way
decomposition of the current term $\rho_1\nabla_p R^1 - \rho_2\nabla_p R^2$
in the proof (see \Cref{app:step5}).
A finer split using Young's inequality with carefully chosen weights
can reduce the factor toward $1$; we keep the factor $3$ to make the
absorption argument transparent.
\end{remark}
Both conditions reduce to $\eps L_R < \lambda$ alone when $R$ is
independent of $p$.

\subsection{Main Theorem}

\begin{theorem}[$L^2$ contraction]\label{thm:main}
Let \Cref{ass:base,ass:mono,ass:lip,ass:bound,ass:unique} hold,
and suppose the GAN-tractable condition \eqref{eq:gan_tractable} holds.
Let $\rho^*$ be the unique Nash equilibrium density of the full MFG.
Then the sequence $\{\rho^k\}$ generated by \Cref{alg:main}
(with exact inner GAN solves) satisfies
\begin{equation}\label{eq:contraction}
  \norm{\rho^{k+1} - \rho^*}_{L^2((0,T)\times\Omega)}
  \;\leq\;
  \kappa\;\norm{\rho^k - \rho^*}_{L^2((0,T)\times\Omega)},
  \qquad
  \kappa = \frac{\eps L_R}{\lambda}.
\end{equation}
\end{theorem}

\begin{corollary}[$\mathcal{W}_2$ convergence]\label{cor:W2}
Under the same assumptions, the densities $\rho^k(t,\cdot)$ converge to
$\rho^*(t,\cdot)$ in the Wasserstein-2 metric uniformly in $t \in [0,T]$.
The rate is geometric:
\[
  \mathcal{W}_2\bigl(\rho^k(t,\cdot),\rho^*(t,\cdot)\bigr)
  \;\leq\; C(\Omega,\rho_{\max})\,\kappa^{k/2},
\]
which follows from \eqref{eq:contraction} and the
$\mathcal{W}_2$-vs-$L^2$ inequality of \Cref{app:wasserstein}.
The factor of $\tfrac{1}{2}$ in the exponent reflects the
$L^1$-to-$\mathcal{W}_2$ square-root scaling and is the best rate
attainable from a pure $L^2$ contraction without sharper transport
estimates.
\end{corollary}

The proof is given below: it proceeds via the
\emph{Lasry-Lions duality identity} (Lemma~\ref{lem:stability} below),
which is the central analytical engine of the paper.
\Cref{app:proofs} contains supplementary expansions, including the
detailed Step~5 absorption and the explicit bilinear-case verification.

\subsection{The Lasry-Lions Stability Lemma}

The key intermediate result is a stability estimate for separable MFGs
under perturbation of the Hamiltonian.

\begin{lemma}[MFG Stability]\label{lem:stability}
Let $(\varphi_1,\rho_1)$ and $(\varphi_2,\rho_2)$ be solutions to two
separable MFG systems sharing the same $f_0$, $\rho_0$, $g$, but with
effective Hamiltonians $H^1(x,p)$ and $H^2(x,p)$ respectively.
Under \Cref{ass:base,ass:mono,ass:bound,ass:pmax} and the second GAN-tractable
condition $3\,\eps\, L_{R,p}\,\rho_{\max} < c_0\,\rho_{\min}$, we have
\begin{equation}\label{eq:stability}
  \lambda\, \norm{\delta\rho}^2_{L^2}
  \;\leq\;
  \eps\, L_R\, \norm{\delta\bar\rho}_{L^2}\, \norm{\delta\rho}_{L^2}
  \;+\; \mathcal{R},
\end{equation}
where $\delta\rho = \rho_1 - \rho_2$, $\delta\bar\rho = \bar\rho_1 - \bar\rho_2$,
and the remainder $\mathcal{R} \leq 0$ from the convexity of $H_0$.
\end{lemma}

\begin{proof}[Proof of Lemma~\ref{lem:stability}]
We apply the \textbf{Lasry-Lions duality trick}: compute the time
derivative of the inner product $\int_\Omega \delta\varphi\cdot\delta\rho\, dx$,
where $\delta\varphi = \varphi_1 - \varphi_2$ and $\delta\rho = \rho_1 - \rho_2$.

\paragraph{Step 1: Compute $\tfrac{d}{dt}\int_\Omega \delta\varphi\cdot\delta\rho\, dx$.}
From HJB$_i$: $\partial_t \varphi_i = -\nu\Delta\varphi_i + H^i(x,\nabla\varphi_i) - f_0(x,\rho_i)$.
From FP$_i$: $\partial_t\rho_i = \nu\Delta\rho_i + \Div(\rho_i\nabla_p H^i(x,\nabla\varphi_i))$.

Subtracting and substituting:
\begin{align}
  \frac{d}{dt}\int_\Omega \delta\varphi\cdot\delta\rho\, dx
  &= \int_\Omega (\partial_t\delta\varphi)\delta\rho\, dx
   + \int_\Omega \delta\varphi(\partial_t\delta\rho)\, dx \notag\\
  &= \underbrace{\int_\Omega \!\bigl[-\nu\Delta\delta\varphi\bigr]\delta\rho
      + \delta\varphi\bigl[\nu\Delta\delta\rho\bigr]dx}_{=\,0\ (\text{Step 2})}
  + \int_\Omega \bigl[H^1(\nabla\varphi_1) - H^2(\nabla\varphi_2)\bigr]\delta\rho\, dx
  \notag\\
  &\quad
  - \int_\Omega \bigl[f_0(\rho_1) - f_0(\rho_2)\bigr]\delta\rho\, dx
  + \int_\Omega \delta\varphi\cdot\Div\bigl(J_1 - J_2\bigr)dx,
  \label{eq:ddt}
\end{align}
where $J_i = \rho_i\nabla_p H^i(\nabla\varphi_i)$ is the agent current.

\paragraph{Step 2: Diffusion terms cancel.}
By integration by parts twice (with homogeneous boundary conditions):
\[
  \int_\Omega (-\nu\Delta\delta\varphi)\delta\rho\, dx
  = \nu\int_\Omega \nabla\delta\varphi\cdot\nabla\delta\rho\, dx
  = -\int_\Omega \delta\varphi(\nu\Delta\delta\rho)\, dx.
\]
The two diffusion contributions exactly cancel.
\textit{(This holds for all $\nu \geq 0$, covering both deterministic and stochastic MFGs.)}

\paragraph{Step 3: Current terms via IBP.}
\[
  \int_\Omega \delta\varphi\cdot\Div(J_1 - J_2)\, dx
  = -\int_\Omega \nabla\delta\varphi\cdot(J_1 - J_2)\, dx.
\]
Denoting $p_i = \nabla\varphi_i$ and $\delta p = p_1 - p_2 = \nabla\delta\varphi$,
equation \eqref{eq:ddt} reduces to:
\begin{equation}\label{eq:reduced}
  \frac{d}{dt}\int_\Omega\delta\varphi\cdot\delta\rho\, dx
  = \underbrace{A + C}_{\text{Hamiltonian}} + \underbrace{B}_{\text{monotonicity}},
\end{equation}
where
\begin{align*}
  A + C &= \int_\Omega \bigl[H^1(p_1) - H^2(p_2)\bigr]\delta\rho
           - \delta p \cdot\bigl[\rho_1\nabla_p H^1(p_1) - \rho_2\nabla_p H^2(p_2)\bigr]\, dx,\\
  B     &= -\int_\Omega \bigl[f_0(\rho_1) - f_0(\rho_2)\bigr]\delta\rho\, dx.
\end{align*}

\paragraph{Step 4: Integrate over $[0,T]$; boundary terms vanish.}
Integrating \eqref{eq:reduced} over $[0,T]$:
\[
  \Bigl[\int_\Omega \delta\varphi\cdot\delta\rho\, dx\Bigr]_0^T
  = \int_0^T (A+C)\, dt + \int_0^T B\, dt.
\]
The left side vanishes:
$\delta\rho(0,\cdot) = 0$ (same initial density),
$\delta\varphi(T,\cdot) = 0$ (same terminal cost $g$).
Hence:
\begin{equation}\label{eq:balance}
  -\int_0^T B\, dt = \int_0^T (A+C)\, dt.
\end{equation}
By \Cref{ass:mono}:
\begin{equation}\label{eq:mono_bound}
  -\int_0^T B\, dt
  = \int_0^T\!\int_\Omega (f_0(\rho_1)-f_0(\rho_2))\delta\rho\, dxdt
  \geq \lambda\,\norm{\delta\rho}^2_{L^2((0,T)\times\Omega)}.
\end{equation}

\paragraph{Step 5: Decompose and bound $A + C$.}
Write $H^i = H_0 + \eps R^i$ where $R^i(x,p) = R(x,\bar\rho_i(x,t),p)$.
Split $A+C$ into base and coupling parts:
\begin{equation}\label{eq:split}
  A+C = \underbrace{(A_0 + C_0)}_{\leq\, 0} + \eps\underbrace{(A_R + C_R)}_{\text{bounded}}.
\end{equation}

\textit{Base part $(A_0 + C_0)$:} Expanding using $H_0$ only,
\begin{align*}
  A_0 + C_0
  &= \int_\Omega \rho_1\bigl[H_0(p_1) - H_0(p_2) - \nabla_p H_0(p_1)\cdot\delta p\bigr]\, dx\\
  &\quad - \int_\Omega \rho_2\bigl[H_0(p_1) - H_0(p_2) - \nabla_p H_0(p_2)\cdot\delta p\bigr]\, dx.
\end{align*}
By \textbf{strong convexity} of $H_0$ (\Cref{ass:base}):
\[
  H_0(p_1) - H_0(p_2) - \nabla_p H_0(p_1)\cdot\delta p \leq -\tfrac{c_0}{2}|\delta p|^2,
  \qquad
  H_0(p_1) - H_0(p_2) - \nabla_p H_0(p_2)\cdot\delta p \geq \tfrac{c_0}{2}|\delta p|^2.
\]
Since $\rho_1,\rho_2 \geq 0$, both terms are non-positive, giving:
\begin{equation}\label{eq:convex_bound}
  A_0 + C_0 \leq -\frac{c_0}{2}\int_\Omega (\rho_1+\rho_2)|\delta p|^2\, dx \leq 0.
\end{equation}

\textit{Coupling part $\eps(A_R + C_R)$:}
The terms involving $R^1 \neq R^2$ split as:
\begin{align}
  \eps(A_R + C_R)
  &= \eps\int_\Omega \bigl[R^1(p_1) - R^2(p_2)\bigr]\delta\rho\, dx
   - \eps\int_\Omega \delta p\cdot\bigl[\rho_1\nabla_p R^1(p_1) - \rho_2\nabla_p R^2(p_2)\bigr]\, dx
  \notag\\
  &= \underbrace{\eps\int_\Omega \bigl[R(x,\bar\rho_1,p_2)
        - R(x,\bar\rho_2,p_2)\bigr]\delta\rho\, dx}_{\text{(i): }\rho\text{-variation}}
  \notag\\
  &\quad + \underbrace{\eps\int_\Omega \bigl[R^1(p_1)-R^1(p_2)\bigr]\delta\rho
        - \delta p\cdot\bigl[\rho_1\nabla_p R^1(p_1) - \rho_2\nabla_p R^2(p_2)\bigr]dx}_{\text{(ii): }p\text{-variation}}.
  \label{eq:coupling_split}
\end{align}

\textit{Bounding term (i):}
By \Cref{ass:lip} (Lipschitz in $\rho$) and Cauchy-Schwarz:
\begin{equation}\label{eq:rho_bound}
  \bigl|\text{(i)}\bigr|
  \leq \eps\, L_R\,\norm{\delta\bar\rho}_{L^2}\,\norm{\delta\rho}_{L^2}.
\end{equation}

\textit{Bounding term (ii):}
A three-way split of the current contribution
$\rho_1\nabla_pR^1(p_1) - \rho_2\nabla_pR^2(p_2)$ together with
\Cref{ass:lip,ass:bound,ass:pmax} (see \Cref{app:step5} for details) yields
\begin{equation}\label{eq:p_bound}
  \bigl|\text{(ii)}\bigr|
  \;\leq\; 3\,\eps\, L_{R,p}\,\rho_{\max}\int_\Omega |\delta p|^2\, dx
  \;+\; \eps\,L_R\,\norm{\delta\bar\rho}_{L^2}\,\norm{\delta\rho}_{L^2}.
\end{equation}

\textit{Absorption of term (ii) into $A_0 + C_0$:}
The non-vacuum condition in \Cref{ass:bound} gives $\rho_1 + \rho_2 \geq 2\rho_{\min}$,
so \eqref{eq:convex_bound} yields the stronger lower bound
\begin{equation}\label{eq:convex_lower}
  A_0 + C_0
  \;\leq\; -\frac{c_0}{2}\int_\Omega (\rho_1+\rho_2)|\delta p|^2\,dx
  \;\leq\; -c_0\,\rho_{\min}\int_\Omega |\delta p|^2\,dx.
\end{equation}
Combining \eqref{eq:p_bound} and \eqref{eq:convex_lower}, and using the
second GAN-tractable condition $3\,\eps\,L_{R,p}\,\rho_{\max} < c_0\,\rho_{\min}$:
\begin{equation}\label{eq:absorbed}
  A_0 + C_0 + \eps(A_R + C_R)_{\text{(ii)}}
  \;\leq\;
  -\bigl(c_0\,\rho_{\min} - 3\,\eps\, L_{R,p}\,\rho_{\max}\bigr)
   \int_\Omega |\delta p|^2\,dx
   \;+\; \eps\,L_R\,\norm{\delta\bar\rho}_{L^2}\,\norm{\delta\rho}_{L^2}
  \;\leq\; \eps\,L_R\,\norm{\delta\bar\rho}_{L^2}\,\norm{\delta\rho}_{L^2}.
\end{equation}
The first bracket is $\geq 0$ by assumption.  The remaining $L_R$-term
combines with term (i) of \eqref{eq:rho_bound} to yield the final
estimate.

\paragraph{Step 6: Conclude.}
Combining \eqref{eq:balance}, \eqref{eq:mono_bound}, \eqref{eq:split},
\eqref{eq:rho_bound}, and \eqref{eq:absorbed}:
\[
  \lambda\,\norm{\delta\rho}^2_{L^2}
  \;\leq\;
  \eps\, L_R\,\norm{\delta\bar\rho}_{L^2}\,\norm{\delta\rho}_{L^2}.
\]
Dividing by $\norm{\delta\rho}_{L^2}$ (assuming non-zero):
\begin{equation}\label{eq:final_stab}
  \boxed{\norm{\delta\rho}_{L^2} \;\leq\; \frac{\eps\, L_R}{\lambda}\,\norm{\delta\bar\rho}_{L^2}
  \;=\; \kappa\,\norm{\delta\bar\rho}_{L^2}.}
\end{equation}
\end{proof}

\subsection{Proof of the Main Theorem and Corollaries}

\begin{proof}[Proof of Theorem~\ref{thm:main}]
Let $\mathcal{T}: L^2 \to L^2$ denote the fixed-point map $\mathcal{T}(\bar\rho) = \rho$,
where $\rho$ is the equilibrium density of the separable MFG with
frozen Hamiltonian $H^{\bar\rho}$. By \Cref{ass:unique}, $\mathcal{T}$ is
well-defined. By Lemma~\ref{lem:stability}, $\mathcal{T}$ is a
$\kappa$-contraction on $L^2$:
\[
  \norm{\mathcal{T}(\bar\rho_1) - \mathcal{T}(\bar\rho_2)}_{L^2}
  \leq \kappa\,\norm{\bar\rho_1 - \bar\rho_2}_{L^2}, \qquad \kappa < 1.
\]
By the \textbf{Banach Fixed-Point Theorem}, $\mathcal{T}$ has a unique fixed
point $\rho^* = \mathcal{T}(\rho^*)$, which is the Nash equilibrium of the
full non-separable MFG. Setting $\bar\rho_2 = \rho^*$ and using
$\rho^{k+1} = \mathcal{T}(\rho^k)$ gives \eqref{eq:contraction}.

For the Wasserstein statement: by the Brenier-Benamou theorem and
the $L^\infty$ bound of \Cref{ass:bound}, $L^2$ convergence of densities
implies $\mathcal{W}_2$ convergence (see \citet{villani2009optimal}, Ch.~7).
\end{proof}

\begin{corollary}[Density-only coupling]\label{cor:density}
If $R(x,\rho,p) = R(x,\rho)$ (independent of $p$), then $L_{R,p} = 0$,
the second condition in \eqref{eq:gan_tractable} is vacuous, and
GAN-tractability reduces to $\eps\, L_R < \lambda$ alone.
\end{corollary}

\begin{corollary}[Bilinear coupling and traffic flow]\label{cor:bilinear}
For the parameterised MFG-LWR family
$H_\eps(x,\rho,p) = \tfrac{1}{2}\norm{p}^2 - p + \eps\,\rho p$ (which
recovers the original LWR Hamiltonian at $\eps = 1$), we have
$R(x,\rho,p) = \rho\cdot p$, so $L_R = u_{\max}$ on the admissible region
$|p| \leq u_{\max}$ (\Cref{ass:pmax}), and $L_{R,p} = \rho_{\max}$.
With $H_0(x,p) = \tfrac{1}{2}\norm{p}^2 - p$ (strong convexity constant $c_0 = 1$)
and the non-vacuum condition $\rho \geq \rho_{\min} > 0$,
the GAN-tractable conditions \eqref{eq:gan_tractable} become:
\begin{equation}\label{eq:traffic_condition}
  \eps\, \frac{u_{\max}}{\lambda} < 1
  \qquad \text{and} \qquad
  3\,\eps\, \frac{\rho_{\max}^2}{\rho_{\min}} < 1.
\end{equation}
The convergence rate is $\kappa = \eps\, u_{\max}/\lambda$.
With standard normalisation $u_{\max} = \rho_{\max} = 1$,
both conditions reduce to $\eps < \min(\lambda,\,\rho_{\min}/3)$.
\end{corollary}

\subsection{Convergence with Approximate Inner Solves}

\begin{proposition}[Inexact inner solves]\label{prop:inexact}
If each inner GAN solve satisfies
$\norm{\rho^{k+1}_N - \rho^{k+1}}_{L^2} \leq \delta_N$, then
\[
  \norm{\rho^k_N - \rho^*}_{L^2}
  \leq \kappa^k\,\norm{\rho^0 - \rho^*}_{L^2} + \frac{\delta_N}{1-\kappa}.
\]
\end{proposition}
\begin{proof}
By triangle inequality and Theorem~\ref{thm:main}:
$\norm{\rho^{k+1}_N - \rho^*} \leq \delta_N + \kappa\,\norm{\rho^k_N - \rho^*}$.
Iterate and sum the geometric series.
\end{proof}

\subsection{Comparison to MFDGM Convergence}

Compared to MFDGM \citep{assouline2023deep}, which converges only in
$L^p$ for $p < 2$ and provides no explicit rate, \Cref{thm:main} is
strictly stronger within the GAN-tractable regime: it gives
$L^2$ and $\mathcal{W}_2$ convergence with the explicit geometric
rate $\kappa^k = (\eps L_R/\lambda)^k$.
The trade-off is that MFDGM applies to arbitrary non-separable
Hamiltonians, whereas \Cref{thm:main} requires the GAN-tractable
condition.
\Cref{tab:comparison} in \Cref{sec:prelim} summarises the comparison
across all three methods.
