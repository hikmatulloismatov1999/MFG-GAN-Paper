\section{Weakly Non-Separable Hamiltonians}
\label{sec:weak}

\subsection{Definition}

\begin{definition}[Weakly Non-Separable Hamiltonian]
\label{def:weak_nonsep}
A Hamiltonian $H : \Omega \times \R \times \R^d \to \R$ is called
\emph{weakly non-separable with coupling strength $\eps \geq 0$} if
\begin{equation}\label{eq:decomp}
  H(x,\rho,p) = \underbrace{H_0(x,p) - f_0(x,\rho)}_{\text{separable base}}
  + \eps\, \underbrace{R(x,\rho,p)}_{\text{coupling residual}},
\end{equation}
where $H_0$ is convex in $p$, $f_0$ is monotone in $\rho$ with constant
$\lambda > 0$, and $R$ is Lipschitz in $\rho$ with constant $L_R$,
uniformly in $(x,p)$.
The case $\eps = 0$ recovers a standard separable MFG.
\end{definition}

\begin{definition}[Simplified density-only tractable regime]
\label{def:tractable}
A weakly non-separable Hamiltonian is in the \emph{GAN-tractable regime} if
\begin{equation}\label{eq:condition}
  \eps\, L_R < \lambda.
\end{equation}
\end{definition}

\begin{remark}
In the density-only setting, condition \eqref{eq:condition} balances coupling
strength ($\eps L_R$) against the restoring force of the monotone interaction
($\lambda$). It is analogous to a spectral gap condition in the theory of
contractive fixed-point iterations.
\end{remark}

\begin{remark}[Two versions of the GAN-tractable condition]
\label{rem:two_versions}
\Cref{def:tractable} is the simplified condition under which the density-only
case contracts at rate $\kappa = \eps L_R/\lambda$.
It is exact when the coupling residual $R$ is independent of the momentum $p$.
For general $R(x,\rho,p)$, the full convergence analysis in
\Cref{sec:convergence} includes both momentum variation and the dependence of
$\nabla_pR$ on the frozen density.  The resulting contraction factor
$\kappa_\eps$ is more conservative than $\eps L_R/\lambda$, but it reduces to
the clean rate when $L_{p\rho}=0$.
We use \Cref{def:tractable} in this section for clarity of exposition;
the formal main theorem is stated under \Cref{def:tractable_full}.
\end{remark}

\subsection{The Traffic Flow Hamiltonian}
\label{sec:traffic}

The MFG-LWR traffic model \citep{huang2019mfg} is the canonical
non-separable benchmark.  Its original Hamiltonian is
\[
  H_{\mathrm{LWR}}(x,\rho,p) = \tfrac{1}{2}\norm{p}^2 - (1-\rho)\,p,
\]
which corresponds to the choice $\eps=1$ in the decomposition
\eqref{eq:decomp} with
\[
  H_0(x,p) = \tfrac{1}{2}\norm{p}^2 - p,
  \qquad
  R(x,\rho,p) = \rho\cdot p,
  \qquad
  f_0(x,\rho) = f_0(\rho).
\]
For the rest of this paper we study the parameterised \emph{family}
\begin{equation}\label{eq:Heps}
  H_\eps(x,\rho,p) \;\coloneqq\; \tfrac{1}{2}\norm{p}^2 - p + \eps\,\rho\,p,
  \qquad \eps \in [0,\infty),
\end{equation}
so that $H_0$ is a separable MFG, $H_1 = H_{\mathrm{LWR}}$ recovers the
original traffic model, and $\eps$ is the experimental control parameter
sweep used in \Cref{sec:experiments}.
By \Cref{ass:pmax} the admissible region is $|p|\leq u_{\max}$, on which
$L_R = u_{\max}$ (since $|R(\rho_1,p)-R(\rho_2,p)|=|p||\rho_1-\rho_2|$
and speeds are physically bounded).

\begin{proposition}[Bilinear exactness and tight rate]
\label{prop:bilinear_exact}
Let $R(x,\rho,p) = \rho\cdot p$, so the family
$H_\eps(x,\rho,p) = H_0(x,p) + \eps\,\rho\, p$ with
$H_0(p) = \tfrac{1}{2}|p|^2 - p$ recovers the original MFG-LWR Hamiltonian
at $\eps = 1$.  For this family:
\begin{enumerate}
  \item The effective Hamiltonian
        $H^k(x,p) = H_0(x,p) + \eps\,\rho^k(x,t)\,p$
        is the \emph{exact} (not approximated) Hamiltonian for the
        subproblem at iteration $k$; no Taylor-expansion error is
        introduced.
  \item The convergence rate of \Cref{thm:main}, evaluated at the no-drift
        uniform equilibrium, is \emph{sharp}: the linearised outer map
        $\mathcal{T}$ has operator norm $\eps\,|1-\eps|/\lambda$,
        which equals $\eps L_R(\rho^*,p^*)/\lambda$ with
        $L_R(\rho^*,p^*) = |p^*| = |1-\eps|$ (\Cref{ass:lip,ass:pmax}).
\end{enumerate}
\end{proposition}
\begin{proof}
\textit{Linearity in $\rho$.} Immediate from $R = \rho p$.

\textit{Sharpness.}  Consider the ergodic MFG on the torus
$\Omega = \mathbb{T}^d$ with $f_0(\rho) = \lambda\rho$, $\nu = 0$, and
$H_\eps$ as above.  The no-drift uniform equilibrium has
$\rho^* \equiv 1$ and constant momentum $p^*$ satisfying
$\nabla_p H_\eps(\rho^*,p^*) = p^* - 1 + \eps = 0$, i.e.\ $p^* = 1 - \eps$.

Linearise about $(\rho^*,p^*)$.  Write $\rho = 1+\delta\rho$,
$\bar\rho = 1+\delta\bar\rho$, $\nabla\varphi = p^* + \delta p$ with
$\delta p = \nabla(\delta\varphi)$.  The relevant partial derivatives at
$(1,p^*)$ are
\[
  \partial_p H_\eps = 0,\qquad
  \partial_\rho H_\eps = \eps\,p^* = \eps(1-\eps),\qquad
  \partial_{pp}^2 H_\eps = 1,\qquad
  \partial_{p\rho}^2 H_\eps = \eps.
\]
The first-order linearised system is therefore
\begin{align*}
  \text{(HJB)}\quad &
  \nu\Delta(\delta\varphi) - \eps(1-\eps)\,\delta\bar\rho
  + \lambda\,\delta\rho = 0, \\
  \text{(FP)}\quad &
  -\nu\Delta(\delta\rho) - \Delta(\delta\varphi)
  - \eps\,\partial_x(\delta\bar\rho) = 0.
\end{align*}
Take Fourier modes on $\mathbb{T}^d$.  For $\nu = 0$ the FP equation
forces $\Delta(\delta\varphi) = -\eps\,\partial_x(\delta\bar\rho)$,
which after substitution into HJB gives
\[
  \lambda\,\delta\rho = \eps(1-\eps)\,\delta\bar\rho
  \quad\Longrightarrow\quad
  \delta\rho = \frac{\eps(1-\eps)}{\lambda}\,\delta\bar\rho
  \;=\; \frac{\eps\,L_R(\rho^*,p^*)}{\lambda}\,\delta\bar\rho.
\]
The signed factor $\eps(1-\eps)$ has absolute value
$\eps|1-\eps| = \eps L_R(\rho^*,p^*)$, so the operator norm of the
linearised map equals $\eps L_R(\rho^*,p^*)/\lambda$ exactly.
Hence, in this linearized regime,
$\|\rho^k-\rho^*\|_{L^2} = (\eps L_R(\rho^*,p^*)/\lambda)^k
\|\rho^0-\rho^*\|_{L^2}$ for small perturbations.  Thus the local
traffic rate used in \Cref{sec:experiments} is attained exactly.
\end{proof}

\begin{remark}[Role of the interaction term]
\label{rem:f0_traffic}
The original MFG-LWR model of \citet{huang2019mfg} contains no explicit
congestion interaction $f_0$ (congestion enters only through $R$).
In this case $\lambda = 0$ and the GAN-tractable condition \eqref{eq:condition}
is not directly satisfied.
We handle this in two ways.
First, one can add a small regularization $f_0(\rho) = \sigma\rho$ for
$\sigma > 0$, which is physically natural (penalizing high density)
and yields $\lambda = \sigma$.
The GAN-tractable condition then becomes $\eps\, u_{\max} < \sigma$,
satisfied for sufficiently strong regularization.
Second, for the pure traffic flow case ($f_0 = 0$), the contraction argument
can be adapted using the strong convexity of $H_0$ alone
(see \Cref{cor:bilinear}), recovering a contraction under
$\eps\, \rho_{\max} < c_0 = 1$.
\end{remark}

\subsection{Position Relative to Physics-Informed Methods}

The following proposition summarizes the comparison within the regime covered
by the outer-map theorem.  It should not be read as an empirical dominance
claim: a full neural implementation and pipeline-level benchmarking are left
for future work.

\begin{proposition}[Distributional outer-map advantage]
\label{prop:gap}
In the GAN-compatible regime of \Cref{def:tractable_full}, the fixed-point
reduction satisfies:
\begin{enumerate}
  \item The exact outer map converges in $L^2$, and hence in $\mathcal{W}_2$
    by the transport estimate in \Cref{cor:W2}.
  \item The equilibrium density $\rho^*$ is learned distributionally
    if the separable inner solver is implemented by a generator-based MFG
    method.
  \item The exact outer-map convergence rate is explicit.  In the density-only
    case it is $(\eps L_R/\lambda)^k$; in the general case it is
    $\kappa_\eps^k$ as in \Cref{thm:main}.
\end{enumerate}
Outside this weak-coupling regime, residual/PINN methods such as MFDGM may be
more appropriate, since the outer fixed-point map need not contract.
\end{proposition}

Items (1) and (3) follow from \Cref{thm:main}; item (2) is conditional on using
a distributional adversarial inner solver.

\begin{proposition}[Necessity of the GAN-tractable condition]
\label{prop:necessity}
In the density-only subcase, the condition $\eps L_R < \lambda$ is not merely
sufficient.  There exists a family of weakly non-separable MFGs parameterized by
$\eps$ such that, for every $\eps L_R \geq \lambda$, the linearisation
of the fixed-point outer map at the equilibrium $\rho^*$
has operator norm $\geq 1$.
Consequently, for every starting density in a neighbourhood of $\rho^*$
with $\rho^0 \neq \rho^*$, the linearised iteration does not converge:
it stagnates when $\eps L_R = \lambda$ and grows geometrically when
$\eps L_R > \lambda$.
\end{proposition}
\begin{proof}
Consider the ergodic MFG on the torus $\Omega = \mathbb{T}^d$ with
$H_0(p) = \tfrac{1}{2}\norm{p}^2$, $f_0(\rho) = \lambda\rho$,
$R(x,\rho) = L_R\rho$ (density-only, $\nu = 0$):
\begin{align*}
  \tfrac{1}{2}\norm{\nabla\varphi}^2 - \lambda\rho + \eps L_R\bar\rho &= c
  \quad\text{(ergodic HJB)},\\
  \Div(\rho\nabla\varphi) &= 0
  \quad\text{(stationary FP)}.
\end{align*}
The unique equilibrium is $\rho^* \equiv 1$, $\nabla\varphi^* = 0$.
Linearizing at $\rho^* = 1$: the stationary FP forces
$\Delta(\delta\varphi) = 0$, hence $\nabla(\delta\varphi) = 0$.
The linearized HJB then reduces to the scalar equation
$-\lambda\,\delta\rho + \eps L_R\,\delta\bar\rho = 0$, giving
\[
  \mathcal{T}(1 + \delta\bar\rho) = 1 + \frac{\eps L_R}{\lambda}\,\delta\bar\rho.
\]
The outer iteration satisfies
$\|\rho^k - \rho^*\|_{L^2} = (\eps L_R/\lambda)^k\|\rho^0 - \rho^*\|_{L^2}$.
When $\eps L_R \geq \lambda$, the factor $(\eps L_R/\lambda)^k \geq 1$
for all $k$, so the error does not decrease.
For $\eps L_R > \lambda$, it grows without bound.
\end{proof}

\begin{remark}
The same ergodic example shows that the density-only contraction rate
$\kappa = \eps L_R/\lambda$ is \emph{sharp}: the linearized iteration achieves
equality $\|\rho^k - \rho^*\| = \kappa^k\|\rho^0 - \rho^*\|$.
Together with \Cref{prop:necessity}, this means the GAN-tractable condition
$\eps L_R < \lambda$ is both necessary and sufficient for linear convergence
of the density-only outer iteration, at least in the linearized
(small-perturbation) regime.
\end{remark}
