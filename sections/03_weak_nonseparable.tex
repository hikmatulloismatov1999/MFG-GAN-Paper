\section{Weakly Non-Separable Hamiltonians}
\label{sec:weak}

\subsection{Definition and Examples}

\begin{definition}[Weakly Non-Separable Hamiltonian]
\label{def:weak_nonsep}
A Hamiltonian $H : \Omega \times \R \times \R^d \to \R$ is called
\emph{weakly non-separable with coupling strength $\eps \geq 0$} if
\begin{equation}\label{eq:decomp}
  H(x,\rho,p) = \underbrace{H_0(x,p) - f_0(x,\rho)}_{\text{separable base}}
  + \eps\, \underbrace{R(x,\rho,p)}_{\text{coupling residual}},
\end{equation}
where:
\begin{enumerate}
  \item[(i)] $H_0 : \Omega \times \R^d \to \R$ is convex in $p$,
  \item[(ii)] $f_0 : \Omega \times \R \to \R$ is monotone increasing in $\rho$
    with constant $\lambda > 0$, i.e.,
    $(f_0(x,\rho_1) - f_0(x,\rho_2))(\rho_1-\rho_2) \geq \lambda(\rho_1-\rho_2)^2$,
  \item[(iii)] $R : \Omega \times \R \times \R^d \to \R$ is Lipschitz in $\rho$
    with constant $L_R$, uniformly in $(x,p)$.
\end{enumerate}
\end{definition}

\noindent The separability limit $\eps = 0$ recovers standard separable MFGs
(APAC-Net regime). The fully non-separable case corresponds to $\eps = 1$.

\begin{definition}[GAN-Tractable Regime]
\label{def:tractable}
A weakly non-separable Hamiltonian is in the \emph{GAN-tractable regime} if
\begin{equation}\label{eq:condition}
  \eps\, L_R < \lambda.
\end{equation}
\end{definition}

\begin{remark}
Condition \eqref{eq:condition} quantifies the balance between
the non-separable coupling strength ($\eps L_R$) and the
restoring force of the monotone interaction ($\lambda$).
It is analogous to a \emph{spectral gap} condition in fixed-point theory.
\end{remark}

\subsection{The Traffic Flow Hamiltonian}

The MFG-LWR traffic model \citep{huang2019mfg} has Hamiltonian
\[
  H(x,\rho,p) = \tfrac{1}{2}\norm{p}^2 - (1-\rho)\,p.
\]
Using \eqref{eq:decomp} with $\eps = 1$:
\begin{align*}
  H_0(x,p) &= \tfrac{1}{2}\norm{p}^2 - p, \quad\quad
  f_0(x,\rho) = 0, \quad\quad
  R(x,\rho,p) = \rho\cdot p.
\end{align*}
Since $R$ is \emph{linear} in $\rho$, the Lipschitz constant is
$L_R = \norm{p}_\infty \leq u_{\max} = 1$ (speeds are physically bounded).

\begin{remark}[Exact Linearization]
For bilinear coupling $R = \rho\cdot p$, the linearization of $R$
around any $\rho^k$ is exact (the residual is zero), which means the
outer fixed-point iteration can in principle converge in fewer steps.
\end{remark}

\subsection{Formal Gap: When GANs Outperform MFDGM}

We now state precisely the regime in which GAN-based methods are superior.

\begin{proposition}[GAN Advantage]
\label{prop:gap}
In the GAN-tractable regime \eqref{eq:condition}:
\begin{enumerate}
  \item[(i)] \textbf{Convergence metric:} Extended-MFGAN converges in
    $\mathcal{W}_2$ (Wasserstein-2), while MFDGM converges in $L^p$ for $p<2$.
    Wasserstein convergence implies convergence in distribution, whereas
    $L^p$ convergence does not.
  \item[(ii)] \textbf{Representation:} Extended-MFGAN learns $\rho$
    distributionally (generator-based), enabling sampling from the equilibrium
    distribution; MFDGM approximates $\rho$ pointwise.
  \item[(iii)] \textbf{Rate:} Extended-MFGAN converges at rate
    $(\eps L_R/\lambda)^k \to 0$, which is explicitly controlled
    by the problem parameters.
\end{enumerate}
Outside the GAN-tractable regime ($\eps L_R \geq \lambda$), MFDGM is preferred
as the fixed-point iteration may diverge.
\end{proposition}

The proof of (i)--(ii) follows from the convergence theorem in
\Cref{sec:convergence}; (iii) is stated there as \Cref{thm:main}.
